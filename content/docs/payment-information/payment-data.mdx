## Payment Data

ClearBank allows you to retrieve payment information against your institution, real accounts, and virtual accounts.
This includes information about payments which have been settled, those that are currently being processed, and those which have been rejected.

<EndpointBlock
  type="get"
  title="Get all payments for your institution"
  endpoints={[
    {
      path: "/v2/Transactions",
      version: "2.0.Sterling"
    },
    {
      path: "/v1/Transactions",
      version: "1.0.Sterling"
    }
  ]}
/>

<EndpointBlock
  type="get"
  title="Get all payments for an account"
  endpoints={[
    {
      path: "/v2/Accounts/{accountId}/Transactions",
      version: "2.0.Sterling"
    }
  ]}
/>

<EndpointBlock
  type="get"
  title="Get a payment for an account"
  endpoints={[
    {
      path: "/v2/Accounts/{accountId}/Transactions/{transactionId}",
      version: "2.0.Sterling"
    }
  ]}
/>

<EndpointBlock
  type="get"
  title="Get all payments for a virtual account"
  endpoints={[
    {
      path: "/v2/Accounts/{accountId}/Virtual/{virtualAccountId}/Transactions",
      version: "2.0.Sterling"
    }
  ]}
/>

<EndpointBlock
  type="get"
  title="Get a payment for a virtual account"
  endpoints={[
    {
      path: "/v2/Accounts/{accountId}/Virtual/{virtualAccountId}/Transactions/{transactionId}",

    }
  ]}
/>

<EndpointBlock
  type="get"

  endpoints={[
    {

  ]}
